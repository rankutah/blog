{{- /* Emits a safe LCP preload link by extracting the hero image and its srcset. Prefers AVIF srcset; falls back to first . When AVIF srcset is found, also emits imagesrcset/imagesizes to help early selection. Only runs on hero/landing/home pages. */ -}} {{- $ctx := . -}} {{- /* Treat plainHome pages (no hero) as non-hero to avoid unnecessary preload */ -}} {{- $plainHome := or $ctx.Params.plainHome false -}} {{- $isHeroPage := or (and $ctx.IsHome (not $plainHome)) (or $ctx.Params.hero (eq $ctx.Params.layout "landing")) -}} {{- /* Also preload when a priority carousel is present on the page */ -}} {{- $hasPriorityCarousel := gt (len (findRE `(?is)data-priority="1"` $ctx.Content)) 0 -}} {{- if or $isHeroPage $hasPriorityCarousel -}} {{- $html := $ctx.Content -}} {{- $avifSourceTags := findRE `(?is)]+type="image/avif"[^>]*>` $html -}} {{- if gt (len $avifSourceTags) 0 -}} {{- $tag := index $avifSourceTags 0 -}} {{- $ssm := findRESubmatch `(?is)\bsrcset="([^"]+)"` $tag -}} {{- $szm := findRESubmatch `(?is)\bsizes="([^"]+)"` $tag -}} {{- $mdm := findRESubmatch `(?is)\bmedia="([^"]+)"` $tag -}} {{- $avifSrcset := "" -}} {{- $avifSizes := "100vw" -}} {{- $media := "" -}} {{- if gt (len $ssm) 0 -}} {{- $avifSrcset = index (index $ssm 0) 1 -}} {{- end -}} {{- if gt (len $szm) 0 -}} {{- $avifSizes = index (index $szm 0) 1 -}} {{- end -}} {{- if gt (len $mdm) 0 -}} {{- $media = index (index $mdm 0) 1 -}} {{- end -}} {{- if ne $avifSrcset "" -}} {{- /* Choose a better preload candidate than the first (often smallest) srcset entry. We pick the smallest width >= target, falling back to the largest available. */ -}} {{- $targetW := 960 -}} {{- $bestUrl := "" -}} {{- $bestW := 0 -}} {{- $largestUrl := "" -}} {{- $largestW := 0 -}} {{- range $i, $raw := split $avifSrcset "," -}} {{- $item := trim $raw " \t\n\r" -}} {{- if ne $item "" -}} {{- $parts := split $item " " -}} {{- $u := "" -}} {{- $wstr := "" -}} {{- if ge (len $parts) 1 -}} {{- $u = trim (index $parts 0) " \t\n\r\"'" -}} {{- end -}} {{- if ge (len $parts) 2 -}} {{- $wstr = trim (index $parts 1) " \t\n\r" -}} {{- end -}} {{- $w := 0 -}} {{- if and $wstr (hasSuffix $wstr "w") -}} {{- $w = int (trim (strings.TrimSuffix "w" $wstr) " ") -}} {{- end -}} {{- if and $u (not (hasPrefix $u "data:")) -}} {{- if gt $w $largestW -}} {{- $largestW = $w -}} {{- $largestUrl = $u -}} {{- end -}} {{- if and (ge $w $targetW) (or (eq $bestW 0) (lt $w $bestW)) -}} {{- $bestW = $w -}} {{- $bestUrl = $u -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- $url := cond (ne $bestUrl "") $bestUrl $largestUrl -}} {{- if and (ne $url "") (not (hasPrefix $url "data:")) -}} {{- end -}} {{- end -}} {{- else -}} {{- $imgMatches := findRESubmatch `(?is)]+src="([^"]+)"` $html -}} {{- $url := "" -}} {{- if gt (len $imgMatches) 0 -}} {{- $url = trim (index (index $imgMatches 0) 1) " \t\n\r\"'" -}} {{- end -}} {{- if and (ne $url "") (not (hasPrefix $url "data:")) -}} {{- end -}} {{- end -}} {{- end -}}
