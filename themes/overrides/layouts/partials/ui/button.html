{{- /* Minimal renderer used by the button shortcode. Expects a plain dict, not a shortcode context. Keys it uses: .text (string) .href (string) .classes (string) .isSubmit (bool) .form (string, optional) */ -}} {{- $text := .text | default "Button" -}} {{- $href := .href | default "" -}} {{- $classes := .classes | default "" -}} {{- $isSubmit:= .isSubmit | default false -}} {{- $form := .form | default "" -}} {{- if $isSubmit -}} {{- else -}} {{ $text }} {{- end -}}
